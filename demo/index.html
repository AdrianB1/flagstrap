HTML

<form role="form">
    <div class="form-group">
        <label>Select Country</label><br>
        <div class="flagstrap"
             data-input-name="country"

Javascript

<script>
    $('.flagstrap').flagStrap();
</script>

Or, with options

<script>
    $('.flagstrap').flagStrap({
        inputName: "country",
        buttonSize: "btn-md",
        buttonType: "btn-default",
        selectedCountry: "DE",
        scrollable: true
    });
</script>

        return this.each(function (i) {

            // Define base elements
            var container = $(this);
            var elementId = 'flagstrap-' + i;

            // Build HTML select and attach to container
            var element = $('<select/>').attr('id', elementId).attr('name', options.inputName);
            container.prepend(element);

            // Check container DIV for options set with data-attributes
            options = $.extend({}, options, container.data());

            // Build HTML Select options, mark selected country if specified in options and  attach to select.
            $.each(countries, function (code, country) {
                var optionAttributes;
                if (options.selectedCountry != undefined) {
                    optionAttributes = {value: code};
                    if (options.selectedCountry == code) {
                        optionAttributes = {value: code, selected: "selected"};
                    }
                }
                element.append($('<option>', optionAttributes).text(country));
            });

            // Begin building bootstrap drop down, yolo!
            var selectedText;
            var selectedLabel;

            // Determine if a country is pre-selected or not and build initial drop down label
            selectedText = element.find('option[value="' + element.val() + '"]').text() || element.find('option').first().text();
            selectedLabel = $('<i/>').addClass('flagstrap-icon flagstrap-' + element.val().toLowerCase()).css('margin-right', options.labelMargin);

            // Build the wrapper
            var wrapper = $('<div/>')
                    .addClass('flagstrap-wrap')
                    .css('display', 'inline-block')
                    .css('position', 'relative')
                    .css('width', '100%');

            element.wrap(wrapper);

            // Construct the drop down button
            var buttonLabel = $('<span/>')
                    .addClass('flagstrap-selected-' + i)
                    .html(selectedLabel)
                    .append(selectedText);

            var button = $('<button/>')
                    .attr('data-toggle', 'dropdown')
                    .addClass('btn ' + options.buttonType + ' ' + options.buttonSize + ' dropdown-toggle')
                    .html(buttonLabel);

            $('<span/>')
                    .addClass('caret')
                    .css('margin-left', '10px')
                    .insertAfter(buttonLabel);

            // Assemble the drop down list items element
            var items = $('<ul/>')
                    .attr('id', elementId)
                    .addClass('dropdown-menu');

            if (options.scrollable) {
                items.css('height', 'auto')
                        .css('max-height', '250px')
                        .css('overflow-x', 'hidden');
            }

            // Insert the drop down button and item list
            element.before(button).before(items);

            // Populate the item list
            element.find('option').each(function () {

                // Get original select option values and labels
                var text = $(this).text();
                var value = $(this).val();

                // Build the flag icon
                var flagIcon = $('<i/>').addClass('flagstrap-icon flagstrap-' + value.toLowerCase()).css('margin-right', options.labelMargin);

                // Build a clickable drop down option item, insert the flag and label, attach click event
                var flagStrapItem = $('<a/>')
                        .attr('data-val', $(this).val())
                        .html(flagIcon)
                        .append(text)
                        .on('click', function () {
                            element.find('option').removeAttr('selected');
                            element.find('option[value="' + $(this).data('val') + '"]').attr("selected", "selected");
                            $('.flagstrap-selected-' + i).html($(this).html());
                            e.preventDefault();
                        });

                // Make it a list item
                var listItem = $('<li/>').prepend(flagStrapItem);

                // Append it to the drop down item list
                $('#' + elementId).append(listItem);

            });

            // Hide the actual HTML select
            element.hide();

        });
